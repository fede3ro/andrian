CREATE TABLE tecnico (
  id int NOT NULL PRIMARY KEY AUTO_INCREMENT,
  cognome varchar(255),
  nome varchar(255),
  targa varchar(255)
);

CREATE TABLE prodotto (
  id int NOT NULL PRIMARY KEY AUTO_INCREMENT,
  nome varchar(255),
  quantita varchar(255),
  stato varchar(255)
);

CREATE TABLE tracciamento (
  id int NOT NULL PRIMARY KEY AUTO_INCREMENT,
  id_prodotto int(255),
  id_tecnico int(255),
  quantita int(255),
  timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

ALTER TABLE tracciamento ADD INDEX index_prodotto (id_prodotto);
ALTER TABLE tracciamento ADD INDEX index_tecnico (id_tecnico);

ALTER TABLE tracciamento ADD CONSTRAINT fk_prodotto_id FOREIGN KEY (id_prodotto) REFERENCES prodotto(id);
ALTER TABLE tracciamento ADD CONSTRAINT fk_tecnico_id FOREIGN KEY (id_tecnico) REFERENCES tecnico(id);
